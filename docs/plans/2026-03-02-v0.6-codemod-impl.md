# v0.6 Migration Tooling — Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Ship a `zodvex migrate` CLI command, a migration guide, and an agent skill so consumers can upgrade from pre-0.5/pre-0.6 zodvex.

**Architecture:** The `migrate` command is added to the existing `zodvex` CLI (`src/cli/`). It reads `.ts`/`.tsx` files in a target directory, applies regex-based identifier renames, updates import specifiers, and reports remaining deprecated usage. The migration guide and agent skill are standalone docs.

**Tech Stack:** TypeScript (Node fs/path), Bun test runner, existing zodvex CLI infrastructure.

```typescript
import fs from 'node:fs'
import path from 'node:path'
import os from 'node:os'
import { migrate } from '../src/cli/migrate'

function createTempDir() {
  return fs.mkdtempSync(path.join(os.tmpdir(), 'zodvex-migrate-'))
}

function writeFile(dir: string, name: string, content: string) {
  const filePath = path.join(dir, name)
  fs.mkdirSync(path.dirname(filePath), { recursive: true })
  fs.writeFileSync(filePath, content)
  return filePath
}

describe('zodvex migrate', () => {
  let tmpDir: string

  beforeEach(() => {
    tmpDir = createTempDir()
  })

  afterEach(() => {
    fs.rmSync(tmpDir, { recursive: true, force: true })
  })

  it('renames Codec-prefixed identifiers to Zodvex prefix', () => {

import type { CodecDatabaseReader } from 'zodvex/server'
import { CodecQueryChain } from 'zodvex/server'

type MyCtx = { db: CodecDatabaseReader }
const chain: CodecQueryChain = db.query('users')
`)

    const result = migrate(tmpDir, { dryRun: false })

    const content = fs.readFileSync(path.join(tmpDir, 'queries.ts'), 'utf-8')
    expect(content).toContain('ZodvexDatabaseReader')
    expect(content).toContain('ZodvexQueryChain')
    expect(content).not.toContain('CodecDatabaseReader')
    expect(content).not.toContain('CodecQueryChain')
    expect(result.filesChanged).toBe(1)

    migrate(tmpDir, { dryRun: false })

    expect(content).toContain('ZodvexDatabaseReader')

    expect(content).toContain('ZodvexQueryChain')
    expect(content).toContain('ZodvexRules')
    expect(content).toContain('ZodvexRulesConfig')
    expect(content).toContain('createZodvexCustomization')
    expect(content).toContain('createBoundaryHelpers')
    expect(content).toContain('BoundaryHelpersOptions')
  })

import { zid } from 'zodvex/core'

const userIdSchema = zid('users')
const postIdSchema = zid("posts")
`)

    migrate(tmpDir, { dryRun: false })

    const content = fs.readFileSync(path.join(tmpDir, 'ids.ts'), 'utf-8')
    expect(content).toContain("zx.id('users')")
    expect(content).toContain('zx.id("posts")')
    expect(content).not.toMatch(/\bzid\(/)

import { zid, zodToConvex } from 'zodvex/core'

const id = zid('users')
`)

    migrate(tmpDir, { dryRun: false })

    const content = fs.readFileSync(path.join(tmpDir, 'imports.ts'), 'utf-8')
    // zid should be removed from import, zx should be added
    expect(content).toContain('zx')
    expect(content).not.toMatch(/import\s*\{[^}]*\bzid\b[^}]*\}/)
    // Usage should be transformed
    expect(content).toContain("zx.id('users')")
  })

  it('adds zx to existing import when not already present', () => {
    writeFile(tmpDir, 'add-zx.ts', `
import { zid, zodToConvex } from 'zodvex/core'

const id = zid('users')
`)

    migrate(tmpDir, { dryRun: false })

    const content = fs.readFileSync(path.join(tmpDir, 'add-zx.ts'), 'utf-8')
    expect(content).toContain('zx')
    expect(content).toContain('zodToConvex')
  })

  it('does not duplicate zx if already imported', () => {
    writeFile(tmpDir, 'has-zx.ts', `
import { zid, zx } from 'zodvex/core'

const id = zid('users')
const date = zx.date()
`)

    migrate(tmpDir, { dryRun: false })

    const content = fs.readFileSync(path.join(tmpDir, 'has-zx.ts'), 'utf-8')
    // Should have exactly one zx in imports
    const zxMatches = content.match(/\bzx\b/g)

    const importLine = content.match(/import\s*\{([^}]*)\}\s*from\s*'zodvex\/core'/)

    const zxInImport = specifiers.match(/\bzx\b/g)
    expect(zxInImport?.length).toBe(1)

    const filePath = writeFile(tmpDir, 'dry.ts', `
import type { CodecDatabaseReader } from 'zodvex/server'
`)

    const result = migrate(tmpDir, { dryRun: true })

    // File should be unchanged
    const content = fs.readFileSync(filePath, 'utf-8')
    expect(content).toContain('CodecDatabaseReader')
    expect(result.filesChanged).toBe(0)
    expect(result.wouldChange).toBe(1)

  it('skips node_modules and non-TS files', () => {

      path.join(tmpDir, 'node_modules/dep/index.ts'), 'utf-8'

  it('reports remaining deprecated usage', () => {

import { zodTable, zQueryBuilder } from 'zodvex/server'

    const result = migrate(tmpDir, { dryRun: false })

    expect(result.remainingDeprecations.length).toBeGreaterThan(0)
    const symbols = result.remainingDeprecations.map(d => d.symbol)
    expect(symbols).toContain('zodTable')
    expect(symbols).toContain('zQueryBuilder')
    expect(symbols).toContain('convexCodec')
    expect(symbols).toContain('mapDateFieldToNumber')
  })

  it('handles Zid type rename to ZxId', () => {
    writeFile(tmpDir, 'types.ts', `

type UserId = Zid<'users'>
`)

    migrate(tmpDir, { dryRun: false })

    // Zid is still exported (deprecated) — no rename needed
    // Just verify the file is not broken
    const content = fs.readFileSync(path.join(tmpDir, 'types.ts'), 'utf-8')
    expect(content).toContain('Zid')
  })

  it('handles files with no zodvex imports (no-op)', () => {
    writeFile(tmpDir, 'unrelated.ts', `

`)

    const result = migrate(tmpDir, { dryRun: false })

    expect(result.filesChanged).toBe(0)
  })
})
```

### Task 2: Implement the migrate function

**Files:**
- Create: `packages/zodvex/src/cli/migrate.ts`

**Step 1: Implement the migrate function**

The function:
1. Walks the target directory for `.ts`/`.tsx` files (skipping `node_modules/`, `.git/`, `_generated/`, `_zodvex/`)
2. For each file, applies rename replacements using string `replaceAll` (not regex where avoidable — these are unambiguous identifiers)
3. Transforms `zid(` → `zx.id(` using word-boundary regex
4. Updates import specifiers: removes `zid` from zodvex imports, adds `zx` if not present
5. In dry-run mode, counts but doesn't write
6. Scans for remaining deprecated symbols and reports them

```typescript
import fs from 'node:fs'
import path from 'node:path'

/** Identifier renames: old → new */
const RENAMES: [string, string][] = [
  ['CodecDatabaseReader', 'ZodvexDatabaseReader'],
  ['CodecDatabaseWriter', 'ZodvexDatabaseWriter'],
  ['CodecQueryChain', 'ZodvexQueryChain'],
  ['CodecRulesConfig', 'ZodvexRulesConfig'],  // Must come before CodecRules
  ['CodecRules', 'ZodvexRules'],
  ['createCodecCustomization', 'createZodvexCustomization'],
  ['createCodecHelpers', 'createBoundaryHelpers'],
  ['CodecHelpersOptions', 'BoundaryHelpersOptions'],
]

/** Deprecated symbols to scan for after migration */
const DEPRECATED_SYMBOLS = [
  'zodTable', 'zodDoc', 'zodDocOrNull',
  'zQueryBuilder', 'zMutationBuilder', 'zActionBuilder',
  'zCustomQueryBuilder', 'zCustomMutationBuilder', 'zCustomActionBuilder',
  'convexCodec', 'mapDateFieldToNumber',
]

const SKIP_DIRS = new Set(['node_modules', '.git', '_generated', '_zodvex', 'dist'])

export type MigrateOptions = {
  dryRun: boolean
}

export type DeprecationWarning = {
  file: string
  line: number
  symbol: string
}

export type MigrateResult = {
  filesScanned: number
  filesChanged: number
  wouldChange: number
  remainingDeprecations: DeprecationWarning[]
}

export function migrate(dir: string, options: MigrateOptions): MigrateResult {
  const result: MigrateResult = {
    filesScanned: 0,
    filesChanged: 0,
    wouldChange: 0,
    remainingDeprecations: [],
  }

  const files = collectFiles(dir)

  for (const filePath of files) {
    result.filesScanned++
    const original = fs.readFileSync(filePath, 'utf-8')
    let content = original

    // Apply identifier renames
    for (const [oldName, newName] of RENAMES) {
      content = content.replaceAll(oldName, newName)
    }

    // Transform zid('x') → zx.id('x') and zid("x") → zx.id("x")
    content = content.replace(/\bzid\(/g, 'zx.id(')

    // Update import specifiers for zid → zx
    content = updateZidImports(content)

    const changed = content !== original

    if (changed) {
      if (options.dryRun) {
        result.wouldChange++
      } else {
        fs.writeFileSync(filePath, content)
        result.filesChanged++
      }
    }

    // Scan for remaining deprecated usage
    scanDeprecations(filePath, content, result.remainingDeprecations)
  }

  return result
}

/** Collect .ts/.tsx files, skipping excluded directories */
function collectFiles(dir: string): string[] {
  const files: string[] = []

  function walk(d: string) {
    for (const entry of fs.readdirSync(d, { withFileTypes: true })) {
      if (entry.isDirectory()) {
        if (!SKIP_DIRS.has(entry.name)) {
          walk(path.join(d, entry.name))
        }
      } else if (entry.isFile() && (entry.name.endsWith('.ts') || entry.name.endsWith('.tsx'))) {
        files.push(path.join(d, entry.name))
      }
    }
  }

  walk(dir)
  return files
}

/**
 * Updates zodvex import specifiers: removes `zid`, adds `zx` if not present.
 *
 * Handles patterns like:
 *   import { zid, zodToConvex } from 'zodvex/core'
 *   → import { zx, zodToConvex } from 'zodvex/core'
 *
 *   import { zid } from 'zodvex'
 *   → import { zx } from 'zodvex'
 */
function updateZidImports(content: string): string {
  // Match import statements from zodvex modules that contain zid
  return content.replace(
    /import\s*\{([^}]*)\}\s*from\s*(['"]zodvex(?:\/\w+)?['"])/g,
    (match, specifiers: string, moduleSpec: string) => {
      if (!/\bzid\b/.test(specifiers)) return match

      // Parse specifiers
      const specs = specifiers.split(',').map(s => s.trim()).filter(Boolean)

      // Remove zid (and type Zid if present — it's still valid but often co-imported)
      const filtered = specs.filter(s => {
        const name = s.replace(/^type\s+/, '').trim()
        return name !== 'zid'
      })

      // Add zx if not already present
      const hasZx = filtered.some(s => s.trim() === 'zx' || s.trim().endsWith(' zx'))
      if (!hasZx) {
        filtered.push('zx')
      }

      return `import { ${filtered.join(', ')} } from ${moduleSpec}`
    }
  )
}

/** Scan file content for remaining deprecated symbol usage */
function scanDeprecations(
  filePath: string,
  content: string,
  warnings: DeprecationWarning[]
) {
  const lines = content.split('\n')
  for (let i = 0; i < lines.length; i++) {
    for (const symbol of DEPRECATED_SYMBOLS) {
      if (new RegExp(`\\b${symbol}\\b`).test(lines[i])) {
        warnings.push({ file: filePath, line: i + 1, symbol })
      }
    }
  }
}
```
